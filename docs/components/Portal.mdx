```tsx title="components/Portal/index.tsx"
import type { FunctionComponent } from 'react'
import { useMemo } from 'react'
import { createPortal } from 'react-dom'

export interface Props {
  elementId: string
}
interface State {}

const Portal: FunctionComponent<Props> = ({ children, elementId }) => {
  const rootElement = useMemo(
    () => document.getElementById(elementId),
    [elementId]
  )
  return createPortal(children, rootElement as Element)
}

export default Portal
```

## Props

| Name                  | Type     | Default |
| --------------------- | -------- | ------- |
| elementId<Required /> | _string_ |         |

## Usage

```tsx
const Page = () => {
  return (
    <div>
      <Portal elementId="portal">
        <div>This area is portal!!</div>
      </Portal>
    </div>
  )
}
```
