텍스트를 복사하는 hooks입니다.

export const useCopyText = () => {
  if (typeof window === 'undefined') return
  if (!window.navigator) return
  return (text) => {
    return new Promise((resolve, reject) => {
      window.navigator.clipboard.writeText(text).then(resolve).catch(reject)
    })
  }
}

export const Button = () => {
  const copyText = useCopyText()
  return (
    <button class="p-1" onClick={() => copyText('Copy!')}>
      Copy!
    </button>
  )
}

```typescript
const useCopyText = () => {
  if (!window.navigator) return

  const copyText = (text: value): Promise<void> => {
    return new Promise((resolve, reject) => {
      window.navigator.clipboard.writeText(text).then(resolve).catch(reject)
    })
  }

  return copyText
}
```

## Usage

```tsx
import { useCopyText } from '../path'

const Page = () => {
  const copyText = useCopyText()

  const onCopy = async () => {
    try {
      await copyText('copy text!')
      window.alert('copying success.')
    } catch (err) {
      console.log(err)
    }
  }

  return (
    <div>
      <button onClick={onCopy}>Copy!</button>
    </div>
  )
}
```
